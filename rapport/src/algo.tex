\begin{algorithm}
  \SetKwInOut{Input}{Entrée}
  \SetKwInOut{Output}{Sortie}

  \Input{Ray, rayon incident.}
  \Input{L, le niveau de récursion, \ie le nombre de fois que le rayon a été
  réfléchi ou réfracté.}
  \Output{La couleur du pixel demandé.}
  ~\\

  \ttt{FinalColor} $\leftarrow$ BLACK\;
  \If {L $>$ MAX\_RECURSION\_LEVEL}{\textbf{return} FinalColor\;}
  ~\\

  \ttt{Record} $\leftarrow$ getClosestHit ( \ttt{Ray} ) \;
  ~\\
 
  \If{Hit}{
    \If{La surface est réfléchissante}{getPixel ( Rayon réfléchi, $L + 1$ ) \;}
    \If{La surface est transparente}{getPixel ( Rayon transmis, $L + 1$ ) \;}
    ~\\

    \ttt{FinalColor} $\leftarrow$ Couleur ambiente + Contribution Lumineuse
    $\times$ (Couleur de l'objet + Couleur du rayon transmis + Couleur du rayon
    réfléchi) \;
  }
  ~\\
  
  \tbf{return} FinalColor \;

  \caption{getPixel\label{alg:getPixel}}
\end{algorithm}
