\newcommand{\customdate}[0]{
  \ifpackageavailable{datetime} 
  {
    \usepackage{datetime}
    \newdateformat{customdateformat}{\monthname[\THEMONTH] \THEYEAR}
    \date{\customdateformat\today}
  }
  {\date{\today}}
}
